81 void OrbitHistory::GetState(const ssfTime& _requestedTime, OrbitState& _returnOrbitState)
82 {
83  // History::GetState returns the index of the nearest state.
84  vector<ssfTime>::difference_type nearestTimeIndex = History::GetState(_requestedTime);
85 
86  int numDataPoints = 0;
87  if(m_OriginalInterpolator)
88  numDataPoints = m_OriginalInterpolator->GetNumberDataPoints();
89  else
90  return;
91  Vector timeVector(numDataPoints);
92  Matrix dataMatrix(numDataPoints, NUM_POSVEL_ELEMENTS);
93  Vector tempVector(NUM_POSVEL_ELEMENTS);
94 
95  // See if there is data at the requested time.
96  if(m_TimeHistory[nearestTimeIndex] != _requestedTime)
97  {
98  // Determine if there is currently an interpolation through the requested point.
99  // if there isn't, create one, then evaluate the interpolation at the requested data point.
100  while(m_OrbitInterpolations.capacity() < nearestTimeIndex)
101  {
102  m_OrbitInterpolations.reserve(m_OrbitHistory.capacity() + HISTORY_RESERVE_SIZE);
103  }
104  // Check to see if the data is initialized to hold an interpolation
105  if(!m_OrbitInterpolations[nearestTimeIndex])
106  {
107  // fill in the 'missing' interpolations
108  for(int kk = m_OrbitInterpolations.size(); kk < nearestTimeIndex; ++kk)
109  m_OrbitInterpolations.push_back(m_OriginalInterpolator->NewPointer());
110 
111  m_OrbitInterpolations.push_back(m_OriginalInterpolator->NewPointer());
112  }
113  // Calculate the interpolation parameters if necessary
114  if(!m_OrbitInterpolations[nearestTimeIndex]->GetValid())
115  {
116  for(int ii = 1; ii <= numDataPoints; ++ii)
117  {
118  timeVector(ii) = m_TimeHistory[nearestTimeIndex - (numDataPoints/2-1) + ii].GetSeconds();
119  dataMatrix(ii,_) = ~m_OrbitHistory[nearestTimeIndex - (numDataPoints/2-1) + ii].GetStateRepresentation()->GetPositionVelocity();
120  }
121 
122  m_OrbitInterpolations[nearestTimeIndex]->Interpolate(timeVector, dataMatrix);
123  }
124

126  _returnOrbitState.SetStateRepresentation(new PositionVelocity);
127  _returnOrbitState.GetStateRepresentation()->SetPositionVelocity(tempVector);
128 
129  // Set the returned orbit frame to the same as the history's stored orbit states.
130  _returnOrbitState.SetOrbitFrame((m_OrbitHistory.front()).GetOrbitFrame());
131  }
