70 void Keplerian::SetPositionVelocity(const Vector& _Position, const Vector& _Velocity)
71 {
72  Vector K_Vector(3); K_Vector(VectorIndexBase+2) = 1;
73  Vector AngMomentum = crossP(_Position, _Velocity);
74  Vector LineNodes = crossP(K_Vector, AngMomentum);
75  double r = norm2(_Position);
76  double v = norm2(_Velocity);
77  double n = norm2(LineNodes);
78  Vector Eccentricity = 1/MU * (pow(v,2) - MU/r) * _Position - _Position.dot(_Velocity)*_Velocity;
79  m_OrbitalElements(ECCENTRICITY) = norm2(Eccentricity);
80  double xi = pow(v,2) / 2 - MU/r;
81 
82  // Semimajor Axis, a
83  m_OrbitalElements(SEMIMAJOR_AXIS) = -MU / (2 * xi);
84 
85  // Inclination, i
86  m_OrbitalElements(INCLINATION) = acos(AngMomentum(VectorIndexBase + 2) / norm2(AngMomentum));
87 
88  // Longitude of the Ascending Node, Omega
89  m_OrbitalElements(LONG_ASC_NODE) = LineNodes(VectorIndexBase) / norm2(LineNodes);
90  if (LineNodes(VectorIndexBase+1) < 0)
91  m_OrbitalElements(LONG_ASC_NODE) = 2*PI - m_OrbitalElements(LONG_ASC_NODE);
92 
93  // Argument of perigee, omega
94  m_OrbitalElements(ARG_PERIGEE) = LineNodes.dot(Eccentricity) / (n * m_OrbitalElements(ECCENTRICITY));
95  if (Eccentricity(VectorIndexBase+2) < 0)
96  m_OrbitalElements(ARG_PERIGEE) = 2*PI - m_OrbitalElements(ARG_PERIGEE);
97 
98  // True Anomaly, nu
99  m_OrbitalElements(TRUE_ANOMALY) = Eccentricity.dot(_Position) / (r * m_OrbitalElements(ECCENTRICITY));
100  if (_Position.dot(_Velocity) < 0)
101  m_OrbitalElements(TRUE_ANOMALY) = 2*PI - m_OrbitalElements(TRUE_ANOMALY);
102 
103  // Special Cases
104  if(m_OrbitalElements(ECCENTRICITY) == 0)
105  { // Circular
106  m_OrbitalElements(ARG_PERIGEE) = 0;
107  if(m_OrbitalElements(INCLINATION) == 0)
108  { // Circular Equatorial - define True Longitude, lambda_true
109  m_OrbitalElements(LONG_ASC_NODE) = 0;
110  m_OrbitalElements(TRUE_ANOMALY) = _Position(VectorIndexBase) / r;
111  if(_Position(VectorIndexBase+1) < 0)
112  m_OrbitalElements(TRUE_ANOMALY) = 2*PI - m_OrbitalElements(TRUE_ANOMALY);
113  }
114  else
115  { // Circular Inclined - define Argument of Latitude, u
116  m_OrbitalElements(TRUE_ANOMALY) = LineNodes.dot(_Position)/(n * r);
117  if(_Position(VectorIndexBase+2) < 0)
118  m_OrbitalElements(TRUE_ANOMALY) = 2*PI - m_OrbitalElements(TRUE_ANOMALY);
119  }
120  }
121  else if(m_OrbitalElements(INCLINATION) == 0)
122  { // Elliptical Equitorial - define True Longitude of Periapsis, ~omega~
123  m_OrbitalElements(LONG_ASC_NODE) = 0;
124  m_OrbitalElements(ARG_PERIGEE) = Eccentricity(VectorIndexBase) / m_OrbitalElements(ECCENTRICITY);
125  if (Eccentricity(VectorIndexBase+2) < 0)
126  m_OrbitalElements(ARG_PERIGEE) = 2*PI - m_OrbitalElements(ARG_PERIGEE);
127  }
128  return;
129 }

183 Vector Keplerian::GetPositionVelocity() const
184 {
185  Vector rv(6);
186  rv(VectorIndexBase + 0) = (GetSemiParameter() * cos(GetTrueAnomaly())) / (1 + GetEccentricity() * cos(GetTrueAnomaly()));
187  rv(VectorIndexBase + 1) = (GetSemiParameter() * sin(GetTrueAnomaly())) / (1 + GetEccentricity() * cos(GetTrueAnomaly()));
188  //rv(VectorIndexBase + 2) = 0;
189  rv(VectorIndexBase + 3) = -sqrt(MU/GetSemiParameter()) * sin(GetTrueAnomaly());
190  rv(VectorIndexBase + 4) = sqrt(MU/GetSemiParameter()) * (GetEccentricity() + cos(GetTrueAnomaly()));
191  //rv(VectorIndexBase + 5) = 0;
192  return rv;

220 {

222  _Position(VectorIndexBase + 1) = (GetSemiParameter() * sin(GetTrueAnomaly())) / (1 + GetEccentricity() * cos(GetTrueAnomaly()));
223  _Position(VectorIndexBase + 2) = 0;

253 double Keplerian::GetEccentricAnomalyFromMeanAnomaly(const double& _MeanAnomaly)
254 {
255 
256  double tolerance = 1e-11; // sets convergence level
258  double _EccentricAnomaly; // returned value
259  double testEccAnomaly; // candidate EA, for convergence test
260  double eccentricity = m_OrbitalElements(ECCENTRICITY);
261  // from Keplerian private data member
262 
263  testEccAnomaly = _MeanAnomaly;
264  _EccentricAnomaly = testEccAnomaly -
265  ( testEccAnomaly - eccentricity*sin(testEccAnomaly) - _MeanAnomaly ) /
266  ( 1 - eccentricity*cos(testEccAnomaly) );
267 
268  while ( fabs(_EccentricAnomaly-testEccAnomaly) > tolerance ) // iterate until _EA = testEA
269  {
270  testEccAnomaly = _EccentricAnomaly;
271  _EccentricAnomaly = testEccAnomaly -
272  ( testEccAnomaly - eccentricity*sin(testEccAnomaly) - _MeanAnomaly ) /
273  ( 1 - eccentricity*cos(testEccAnomaly) );
274  }
275 
276  return _EccentricAnomaly;
277 }

287 void Keplerian::GetTrueAnomalyFromEccentricAnomaly(const double& _EccentricAnomaly)
288 {
289 
290  double cosTrueAnomaly;
291  double sinTrueAnomaly;
292  double semimajoraxis = m_OrbitalElements(SEMIMAJOR_AXIS);
293  double eccentricity = m_OrbitalElements(ECCENTRICITY);
294 
295  cosTrueAnomaly = ( eccentricity - cos(_EccentricAnomaly) ) /
296  ( eccentricity*cos(_EccentricAnomaly) - 1 );
297  sinTrueAnomaly = ( ( semimajoraxis*sqrt(1 - eccentricity*eccentricity) ) /
298  ( semimajoraxis*(1 - eccentricity*cos(_EccentricAnomaly)) ) ) *
299  sin(_EccentricAnomaly);
300 
301  m_OrbitalElements(TRUE_ANOMALY) = atan2( sinTrueAnomaly, cosTrueAnomaly );
302  if ( m_OrbitalElements(TRUE_ANOMALY) < 0 );
303  {
304  m_OrbitalElements(TRUE_ANOMALY) = m_OrbitalElements(TRUE_ANOMALY) + 2*PI;
305  }
306 
307  return;
308 }
