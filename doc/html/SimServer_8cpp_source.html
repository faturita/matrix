415 static Vector AttituteDynamics(const ssfTime &_time, const Vector& _integratingState, Orbit *_Orbit, Attitude *_Attitude, const Matrix &_parameters, const Functor &_forceFunctorPtr)
416 {
417  static Vector stateDot(7);
418  static Matrix bMOI(3,3);
419  static Matrix qtemp(4,3);
420  static Matrix Tmoment(3,1);
421  static Vector qIn(4);
422  static Vector wIn(3);
423  qIn = _integratingState(_(VectorIndexBase,VectorIndexBase + 3));
424  wIn = _integratingState(_(VectorIndexBase + 4,VectorIndexBase + 6));
425  qIn /= norm2(qIn);
426 
427 // _attitudeState->SetState(Rotation(Quaternion(qIn)), wIn); // Need to speed up in some way
428  qtemp(_(VectorIndexBase+0,VectorIndexBase+2),_(VectorIndexBase+0,VectorIndexBase+2)) = skew(qIn(_(VectorIndexBase+0,VectorIndexBase+2))) + qIn(VectorIndexBase+3) * eye(3);
429  qtemp(VectorIndexBase+3, _(VectorIndexBase+0,VectorIndexBase+2)) = -(~qIn(_(VectorIndexBase+0,VectorIndexBase+2)));
430  qtemp(_,MatrixIndexBase) = 0.5 * qtemp * wIn;
431 
432  bMOI = _parameters(_(MatrixIndexBase+0,MatrixIndexBase+2),_(MatrixIndexBase+0,MatrixIndexBase+2));
433  Tmoment(_,_) = (_forceFunctorPtr.Call(_time, _Orbit->GetStateObject(), _Attitude->GetStateObject()))(_);
434  Tmoment = (bMOI.inverse() * (Tmoment - skew(wIn) * (bMOI * wIn)));
435 
436  stateDot(_(VectorIndexBase,VectorIndexBase + 3)) = qtemp(_,MatrixIndexBase);
437  stateDot(_(VectorIndexBase + 4,VectorIndexBase + 6)) = Tmoment(_,MatrixIndexBase);
438 
439  return stateDot;
440 }
