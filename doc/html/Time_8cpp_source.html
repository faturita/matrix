30 ssfJulianDate JulianDate(const ssfSeconds &_time)
31 {
32  time_t t = static_cast<time_t>(floor(_time));
33  tm DateTime = *gmtime(&t);
34  ssfJulianDate JD = 0;
35  double uSec = _time - floor(_time);
36  JD += 367 * (DateTime.tm_year + 1900);
37  JD -= floor((7 * ((DateTime.tm_year + 1900) + floor(((DateTime.tm_mon + 1) + 9)/12))) / 4);
38  JD += floor((275 * (DateTime.tm_mon + 1)) / 9);
39  JD += DateTime.tm_mday;
40  JD += 1721013.5;
41  JD += (((DateTime.tm_sec + uSec)/60.0 + DateTime.tm_min) / 60.0 + DateTime.tm_hour) / 24;
